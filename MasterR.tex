% ⚫ BULLET POINT ENTRY — used for each responsibility/description line
% �� If line spacing between bullet points is too tight or loose, adjust \vspace here
\newcommand{\resumeItem}[1]{
  \item\small{
    {#1 \vspace{-2pt}} % Try -1pt or -3pt if spacing feels off
  }
}

% �� FOUR-FIELD ENTRY (Company, Dates, Role, Location/Details)
% ��️ If vertical gap below company name is too much/little, change \vspace here
\newcommand{\resumeSubheading}[4]{
  \vspace{-2pt}\item % ⬅️ Adjust this to control space before company line
    \begin{tabular*}{0.97\textwidth}[t]{l@{\extracolsep{\fill}}r}
      \textbf{#1} & #2 \\
      \textit{\small#3} & \textit{\small #4} \\
    \end{tabular*}\vspace{-7pt} % ⬅️ Controls space between company line and bullet start
}

% �� COMPACT FORMAT — use when showing Company | Role on one line
% �� If space between this and bullet list is weird, adjust \vspace below
\newcommand{\resumeCompactHeading}[3]{
  \item
  \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
    \textbf{#1} \textbar\ \textit{#2} & \textit{#3} \\
  \end{tabular*}\vspace{-5pt} % ⬅️ Controls spacing below heading
}

% �� SUB-SUBHEADING — for nested rows (rarely used)
\newcommand{\resumeSubSubheading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \textit{\small#1} & \textit{\small #2} \\
    \end{tabular*}\vspace{-7pt}
}

% �� PROJECT HEADING — project name and date
% �� To align project name margin with section heading, tweak left margin globally (see below)
\newcommand{\resumeProjectHeading}[2]{
    \item
    \begin{tabular*}{0.97\textwidth}{l@{\extracolsep{\fill}}r}
      \small#1 & #2 \\
    \end{tabular*}\vspace{-5pt}
}

% �� Single sub-item (used within nested lists)
\newcommand{\resumeSubItem}[1]{\resumeItem{#1}\vspace{-4pt}}

% �� CUSTOM BULLET SYMBOL for deeper list levels (not needed unless nesting)
\renewcommand\labelitemii{$\vcenter{\hbox{\tiny$\bullet$}}$}

% �� EXPERIENCE/EDUCATION/PROJECT SECTION START
% ��️ If section contents look too indented, adjust `leftmargin` (e.g., try 0in or -0.1in)
\newcommand{\resumeSubHeadingListStart}{\vspace{-1pt}\begin{itemize}[leftmargin=0in, label={}]}

% �� SECTION END
% ��️ If too much space below a section, tweak \vspace here
\newcommand{\resumeSubHeadingListEnd}{\end{itemize}\vspace{-10pt}}

% �� BULLET LIST START (for experience/project points)
% �� To reduce indent of bullet points, decrease `leftmargin`
\newcommand{\resumeItemListStart}{\vspace{-2pt}\begin{itemize}[leftmargin=0.3in]} % Try 0.25in or 0.2in if too indented

% �� BULLET LIST END
% �� To reduce space between this and next job/project, make \vspace more negative (e.g., -12pt)
\newcommand{\resumeItemListEnd}{\end{itemize}\vspace{-7pt}}

% �� BULLET LIST END (tighter version)
\newcommand{\resumeItemListEndTight}{\end{itemize}\vspace{-6pt}} % Use when you're running out of space
